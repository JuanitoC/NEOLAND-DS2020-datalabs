# Realizo test para comprobar si `annual_percentage_rate` tiene distribución normal.
# a) Shapiro-Wilk Test
from scipy.stats import shapiro
data = DF['annual_percentage_rate']
stat, p = shapiro(data)
print('stat=%.3f, p=%.3f' % (stat, p))
if p > 0.05:
	print('Shapiro-Wilk Test: Distribución normal')
else:
	print('Shapiro-Wilk Test: Distribución no-normal')

if p > 0.05:
	print('D’Agostino’s K^2 Test: Distribución normal')
